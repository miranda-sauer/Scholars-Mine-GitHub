	Paste the following into the .spec file after running 'pyinstaller --onedir <script name>.py':
	
		import sys
		sys.setrecursionlimit(5000)

	then run 'pyinstaller <script name>.spec'



	OR


	Create a setup.py file with the following code and run 'python setup.py build':

		from cx_Freeze import setup, Executable

		base = None    

		executables = [Executable("<source_name>.py", base=base)]

		packages = ["<all included libraries>"] # i.e. ["idna","pickle"]
		options = {
		    'build_exe': {    
		        'packages':packages,
		    },    
		}

		setup(
    			name = "<give_it_a_name>",
    			options = options,
    			version = '0.11',
    			description = '',
    			executables = executables
		)
